Reverse LinkedList Recursive

Node printReverse(Node head)
{
    if(head==null || head.next==null)
     return head;
    Node reverse = printReverse(head.next);
    head.next.next = head;
    head.next = null;
    return reverse;
}

Reverse LinkedList Iterative

Node reverse(Node node)
    {
        Node prev = null;
        Node current = node;
        Node next = null;
        while (current != null) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
        }
        node = prev;
        return node;
    }
